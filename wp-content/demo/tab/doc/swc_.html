     eventType:"mouseover",
     showType:"slide",
     beforeSwc:function(item){
             // @this 为Swc对象,item为激动项
       console.log(this, item);
     },
     showSwc : function(){
             //this为显示的面板项
       console.log(this);
     },
     hideSwc : function(){
             //this为所有隐藏的面板项
       console.log(this);
     }
   });
